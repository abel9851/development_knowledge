```javascript

function iterativeSum(n) {
  let total = 0;
  for (let i = 1; i <= n; i++) {
    total += i;
  }
  return total;
}
```

```javascript

function recursiveSum(n) {
  // 기본 케이스 (Base case): 멈추는 조건
  if (n <= 0) {
    return 0;
  }
  // 재귀 케이스 (Recursive case): 자기 자신을 호출
  return n + recursiveSum(n - 1);
}
```
